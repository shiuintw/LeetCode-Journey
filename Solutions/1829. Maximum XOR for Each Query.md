# Approach
notice that:
1. `0 ^ a = a`
2. `if a ^ b = c then a ^ c = b`

# Complexity
- Time complexity:
  $$O(n)$$

- Space complexity:
  $$O(n)$$

# Code
```java []
class Solution {
    private int pow(int x, int y) {
        if (y == 0) return 1;
        else if (y % 2 == 1) return pow(x, y - 1) * x;
        else {
            int t = pow(x, y / 2);
            return t * t;
        }
    }
    public int[] getMaximumXor(int[] nums, int maximumBit) {
        int suf = 0;
        int[] ans = new int[nums.length];
        maximumBit = pow(2, maximumBit);
        for (int i = 0; i < nums.length; i++) {
            suf ^= nums[i];
            ans[nums.length - i - 1] = suf ^ (maximumBit - 1);
        }
        return ans;
    }
}
```
