# Approach
1. find the longest consecutive prefix increasing subarray
2. find the longest consecutive suffix increasing subarray
3. use the Monotonic Stack to combine two subarrays.

# Complexity
- Time complexity:
  $$O(Nlog(N))$$

- Space complexity:
  $$extra:O(1)$$
  $$total:O(N)$$

# Code
```java []
class Solution {
    public int findLengthOfShortestSubarray(int[] arr) {
        int pos = 0;
        for (;pos < arr.length - 1; pos++) { // front incresing
            if (arr[pos + 1] < arr[pos]) break;
        }
        int posBack = arr.length - 1;
        for (;posBack > pos; posBack--) { // back increasing
            if (arr[posBack] < arr[posBack - 1]) break;
        }

        if (pos == arr.length - 1) return 0;

        // stack at [0, pos]
        for (int i = posBack; i < arr.length; i++) {
            if (arr[i] >= arr[pos]) break;
            else {
                int t = -1;
                // jump binary search
                for (int jump = pos + 1; jump > 0; jump /= 2) {
                    while (t + jump <= pos && arr[t + jump] <= arr[i]) {
                        t += jump;
                    }
                }
                arr[t + 1] = arr[i];
                posBack++;
            }
        }
        return posBack - pos - 1;
    }
}
```
